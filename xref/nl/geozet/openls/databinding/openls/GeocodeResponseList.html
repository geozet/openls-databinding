1   package nl.geozet.openls.databinding.openls;
2   
3   import java.util.Vector;
4   
5   import nl.geozet.openls.databinding.common.XmlNamespaceConstants;
6   
7   public class GeocodeResponseList {
8       /*
9        * http://schemas.opengis.net/ols/1.2.0/LocationUtilityService.xsd
10       * 
11       * <complexType name="GeocodeResponseListType"> <sequence> <element
12       * name="GeocodedAddress" type="xls:GeocodedAddressType"
13       * maxOccurs="unbounded"> <annotation> <documentation>The list of 1-n
14       * addresses that are returned for each Address request, sorted by
15       * Accuracy.</documentation> </annotation> </element> </sequence> <attribute
16       * name="numberOfGeocodedAddresses" type="nonNegativeInteger"
17       * use="required"> <annotation> <documentation>This is the number of
18       * responses generated per the different requests. Within each geocoded
19       * address tit's possible to have multiple candidates</documentation>
20       * </annotation> </attribute> </complexType>
21       */
22  
23      private Vector<GeocodedAddress> geocodedAddress = new Vector<GeocodedAddress>();
24      private int numberOfGeocodedAddresses;
25  
26      public void addGeocodedAddress(GeocodedAddress val) {
27          geocodedAddress.add(val);
28      }
29  
30      public GeocodedAddress getGeocodedAddressAt(int i) {
31          return (GeocodedAddress) geocodedAddress.get(i);
32      }
33  
34      public int getGeocodedAddressSize() {
35          return geocodedAddress.size();
36      }
37  
38      public void setNumberOfGeocodedAddresses(int val) {
39          numberOfGeocodedAddresses = val;
40      }
41  
42      public int getNumberOfGeocodedAddresses() {
43          return numberOfGeocodedAddresses;
44      }
45  
46      public String toXML() {
47          String xml = "<" + XmlNamespaceConstants.OPENLS_NAMESPACE_PREFIX
48                  + ":GeocodeResponseList " + "numberOfGeocodedAddresses=\""
49                  + Integer.toString(getGeocodedAddressSize()) + "\">";
50          for (GeocodedAddress gca : geocodedAddress) {
51              xml += gca.toXML();
52          }
53          xml += "</" + XmlNamespaceConstants.OPENLS_NAMESPACE_PREFIX
54                  + ":GeocodeResponseList>";
55          return xml;
56      }
57  }
